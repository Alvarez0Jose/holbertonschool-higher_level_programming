def test_add_integer():
    """
    Testing cases for add_integer function
    """

    # Test valid integer additions

    assert add_integer(1, 2) == 3
    assert add_integer(-5, 10) == 5
    assert add_integer(0, 0) == 0

    # Test valid float additions

    assert add_integer(2.5, 3,7) == 6
    assert add_integer(0.1, 0.2) == 0

    # Test casting float to integer

    assert add_integer(2.9, 3) == 5
    assert add_integer(5, 4.1) == 9

    # Test invalid input

    try:
        add_integer("2", 3)
    expect TypeError as e:
        assert str(e) == "a must be an integer or b must be an integer"

    try:
        add_integer(2, "3")
    expect TypeError as e:
        assert str(e) == "a must be an integer or b must be an integer"

    try:
        add_integer("2", "3")
    expect TypeError as e:
        assert str(e) == "a must be an integer and b must be an integer"

test_add_integer()
